# Question 2.a) Qual é o número máximo de saltos que um passageiro pode fazer em uma única viagem entre
# duas cidades atendidas?

# Aqui entendi que era para calcular o maior caminho entre as possibilidades de saltos entre duas cidades destintas
# Como se fosse percorrer o maior número de cidades sem voltar para a mesma

originCities = G.nodes()
destinationCities = G.nodes()
origin = ''
destination = ''

size = 0
for originCity in originCities:
    for destinationCity in destinationCities:
        for path in nx.all_simple_paths(G, originCity, destinationCity):
            if size < len(path):
                size = len(path)
                biggerPath = path
                origin = originCity
                destination = destinationCity

print("Largest Path between two cities: ")
print("\tOrigin: "+origin)
print("\tDestination: "+destination)
print("\tSize: "+ str(size))
print("\tBigger Path: "+ str(biggerPath))
